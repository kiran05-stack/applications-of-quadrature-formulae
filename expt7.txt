kill(all)$

fpprintprec:4$

/* Define function */
f(x) := exp(x)$

/* Lower and upper limits */
x0: 0$
xn: 1$
n: 5$ /* Number of intervals */
h: (xn - x0) / n$

/* Arrays for x and y */
array(x, n+1)$
array(y, n+1)$

s: 0$

/* Compute x[i] and y[i] = f(x[i]) */
for i:0 thru n do (
  x[i]: x0 + i*h,
  y[i]: f(x[i]),
  print("x[", i, "] = ", x[i], " , y[", i, "] = ", y[i])
)$

/* Sum of middle y[i] terms */
for i:1 thru n-1 do (
  s: s + y[i]
)$

/* Trapezoidal Rule */
I: (h/2)*(y[0] + y[n] + 2*s)$

print("The value of the integral using Trapezoidal Rule is: ", I)$

/* Compute x[i] and y[i] */

  x[i]: x0 + i * h,
  y[i]: f(x[i]),
  print("x[", i, "] = ", x[i], " , y[", i, "] = ", y[i])
)$

/* Apply Simpson's 1/3 rule */
for i: 1 thru n - 1 do (
  if remainder(i, 2) = 0 then s2: s2 + y[i] else s1: s1 + y[i]
)$
